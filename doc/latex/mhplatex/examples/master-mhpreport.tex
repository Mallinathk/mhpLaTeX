%% Abstract:
%%   LaTeX master file master-mhpreport.tex serves as a template for
%% documents based on the mhpreport document class of the mhpLaTeX
%% document preparation framework.
%%
%% Details:
%% - Processing this LaTeX master file requires the pdfTeX engine (with
%%   the ε-TeX extensions enabled).
%% - Commands for document generation:
%%   1. $ pdflatex master-mhpreport
%%        # Generate the output document in PDF format by calling pdfTeX
%%        # in PDF mode. In this case, the pdftex output driver should
%%        # be specified as a document class option.
%%      $ latex master-mhpreport && dvipdfmx master-mhpreport
%%        # Alternatively, generate the output document in PDF format by
%%        # calling pdfTeX in DVI mode. In this case, the dvipdfmx
%%        # output driver should be specified as a document class
%%        # option. This command is an alternative to the preceding
%%        # command, allowing to generate the output document in a way
%%        # closer to how it would be generated by means of the original
%%        # TeX engine. Note that this may imply a significantly smaller
%%        # PDF file size.
%%   2. $ makeglossaries master-mhpreport
%%        # Generate LaTeX code for formatting the glossary sections.
%%        # Which index processor is internally called by the
%%        # makeglossaries program depends on the value of the
%%        # glossaries/processor document class option. It can be
%%        # ‘makeindex’ or ‘xindy’.
%%   3. $ bibtex master-mhpreport
%%        # Generate LaTeX code for formatting the bibliography section
%%        # by means of the bibtex program. In this case, the
%%        # biblatex/backend document class option should be set to
%%        # ‘bibtex’.
%%      $ bibtex8 -c <csfile> -W master-mhpreport
%%        # Alternatively, generate LaTeX code for formatting the
%%        # bibliography section by means of the bibtex8 program. In
%%        # this case, the biblatex/backend document class option should
%%        # be set to ‘bibtex8’. <csfile> denotes a BibTeX character set
%%        # and sort definition file such as ascii.csf, latin1.csf or
%%        # latin9.csf. This should match the encoding of the
%%        # bibliographic database files.
%%      $ biber master-mhpreport
%%        # Alternatively, generate LaTeX code for formatting the
%%        # bibliography section by means of the biber program. In this
%%        # case, the biblatex/backend document class option should be
%%        # set to ‘biber’.
%%

\documentclass[
  %% output driver
  %% --> pdftex, dvipdfmx
  pdftex,
  %% final/draft document version?
  %% --> final, draft
  draft,
  %% print/screen document version?
  %% --> print, screen
  screen,
  %% main document language
  %% --> english, USenglish, UKenglish, ngerman, naustrian
  UKenglish,
  %% paper size
  %% --> an ISO 216 paper size, a North American paper size,
  %%     portrait, landscape
  paper=a4,
  %% font size
  %% --> in pt
  fontsize=11pt,
  %% DIV factor
  %% --> default, calc, an integer ≥ 4
  DIV=default,
  %% binding correction
  %% --> in mm
  BCOR=0mm,
  %% glossary section headings in the table of contents?
  %% --> nottotoc, totoc, totocnumberline
  glossaries=totoc,
  %% acronym style
  %% --> default, dua, footnote, smallcaps, smaller, description,
  %%     description+dua, description+footnote
  glossaries/acronymstyle=default,
  %% index processor used along with the glossaries package
  %% --> default, makeindex, xindy
  glossaries/processor=makeindex,
  %% bibliography–citation style
  %% --> default, a style known to the biblatex package
  biblatex/style=default,
  %% bibliographic database backend used along with the biblatex package
  %% --> default, bibtex, bibtex8, biber
  biblatex/backend=bibtex8,
  %% encoding of the bibliographic database files
  %% --> default, auto, x-ascii, x-iso-8859-1, x-iso-8859-15, utf8,
  %%     another encoding known to the inputenx package
  biblatex/bibencoding=x-ascii,
]{mhpreport}[2013/07/14]
